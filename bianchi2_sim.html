program bianchi2_sim

        ! Uses
    use s2_types_mod
    use s2_sky_mod, only: S2_SKY_FILE_TYPE_MAP, S2_SKY_FILE_TYPE_SKY
    use bianchi2_sky_mod
    use bianchi2_error_mod
    use pix_tools, only: nside2npix
    use extension, only: getArgument, nArguments
    use paramfile_io, only: paramfile_handle, parse_init, parse_int,     parse_real, parse_double, parse_lgt, parse_string, concatnl

        ! Calls
    call bianchi2_error
    call bianchi2_sky_apply_beam
    call bianchi2_sky_compute_alm
    call bianchi2_sky_free
    call bianchi2_sky_param_write
    call bianchi2_sky_rotate
    call bianchi2_sky_write
    call getArgument

        ! Variables
    character (len=S2_STRING_LEN) :: filename_param
    character (len=S2_STRING_LEN) :: description
    character (len=S2_STRING_LEN) :: line
    type (paramfile_handle) :: handle
    real (kind=s2_dp), parameter :: OMEGA_MATTER_LOWER = 0d0
    real (kind=s2_dp), parameter :: OMEGA_MATTER_UPPER = 1d0
    real (kind=s2_dp), parameter :: OMEGA_MATTER_DEFAULT = 0.3d0
    real (kind=s2_dp), parameter :: OMEGA_LAMBDA_LOWER = 0d0
    real (kind=s2_dp), parameter :: OMEGA_LAMBDA_UPPER = 1d0
    real (kind=s2_dp), parameter :: OMEGA_LAMBDA_DEFAULT = 0.6d0
    real (kind=s2_dp), parameter :: H_LOWER = 0d0
    real (kind=s2_dp), parameter :: H_UPPER = 10d0
    real (kind=s2_dp), parameter :: H_DEFAULT = 1d-2
    real (kind=s2_dp), parameter :: ZE_LOWER = 1d2
    real (kind=s2_dp), parameter :: ZE_UPPER = 1d4
    real (kind=s2_dp), parameter :: ZE_DEFAULT = 1d3
    real (kind=s2_dp), parameter :: WH_DEFAULT = 1d-10
    real (kind=s2_sp), parameter :: FWHM_DEFAULT = 330d0
    logical, parameter :: RHAND_DEFAULT = .true.
    character (len=*), parameter :: FILE_TYPE_MAP_STR = 'map'
    character (len=*), parameter :: FILE_TYPE_SKY_STR = 'sky'
    character (len=S2_STRING_LEN) :: filename_out
    character (len=S2_STRING_LEN) :: filetype_str = FILE_TYPE_MAP_STR
    integer :: filetype = S2_SKY_FILE_TYPE_MAP
    type (bianchi2_sky) :: b
    real (kind=s2_dp) :: omega_matter
    real (kind=s2_dp) :: omega_lambda
    real (kind=s2_dp) :: h
    real (kind=s2_dp) :: zE
    real (kind=s2_dp) :: wH
    integer :: nside
    integer :: lmax
    real (kind=s2_sp) :: alpha = 0e0
    real (kind=s2_sp) :: beta = -90e0
    real (kind=s2_sp) :: gamma = 0e0
    logical :: rhand = .true.
    logical :: apply_beam = .false.
    real (kind=s2_sp) :: fwhm = FWHM_DEFAULT

end program bianchi2_sim
